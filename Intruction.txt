-- =========================================================
-- THE BRAIN: GOD-TIER COMBAT (Silent Aim & Smart Dodge)
-- =========================================================

-- 1. Утилита для поиска ближайшей жертвы (Silent Aim)
local function getBestTarget()
    local target = nil
    local shortestDistance = 1000 -- Максимальный радиус захвата

    for _, player in pairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character and player.Character:FindFirstChild("Humanoid") then
            -- Проверка: жив ли враг и не в нашей ли он команде
            if player.Character.Humanoid.Health > 0 and (player.Team ~= LocalPlayer.Team) then
                local head = player.Character:FindFirstChild("Head")
                if head then
                    -- Считаем дистанцию от нас до его головы
                    local mag = (head.Position - HumanoidRoot.Position).Magnitude
                    if mag < shortestDistance then
                        -- Проверка на стены (Raycast): пуля летит только если врага видно
                        local rayParams = RaycastParams.new()
                        rayParams.FilterType = Enum.RaycastFilterType.Exclude
                        rayParams.FilterDescendantsInstances = {Character, workspace.CurrentCamera}
                        
                        local rayResult = workspace:Raycast(Camera.CFrame.Position, (head.Position - Camera.CFrame.Position).Unit * mag, rayParams)
                        
                        -- Если луч попал во врага или ничего не встретил на пути — это цель!
                        if not rayResult or rayResult.Instance:IsDescendantOf(player.Character) then
                            target = head
                            shortestDistance = mag
                        end
                    end
                end
            end
        end
    end
    return target
end

-- 2. ВЗЛОМ ДВИЖКА (Silent Aim через Metatable Hook)
-- Мы перехватываем момент, когда игра спрашивает "куда попала пуля?"
local oldNamecall
oldNamecall = hookmetamethod(game, "__namecall", function(self, ...)
    local method = getnamecallmethod()
    local args = {...}

    -- Если кнопка включена и игра делает проверку выстрела (Raycast)
    if Settings.Features.SilentAim and (method == "Raycast" or method == "FindPartOnRayWithIgnoreList") then
        local goal = getBestTarget()
        if goal then
            -- Если цель найдена, "подрисовываем" траекторию пули прямо в голову
            if method == "Raycast" then
                -- Для новых систем (Raycast)
                return {Instance = goal, Position = goal.Position, Material = goal.Material, Normal = Vector3.new(0,1,0)}
            else
                -- Для старых систем (FindPartOnRay)
                return goal, goal.Position, goal.CFrame.LookVector, goal.Material
            end
        end
    end
    return oldNamecall(self, ...)
end)

-- 3. СИСТЕМА УКЛОНЕНИЯ (Smart Dodge через Heartbeat)
-- Работает в физическом цикле, чтобы успеть среагировать на пулю
RunService.Heartbeat:Connect(function()
    if not Settings.Features.SmartDodge or not HumanoidRoot then return end

    -- Сканируем всё, что летит в Workspace
    for _, projectile in pairs(workspace:GetChildren()) do
        -- Если объект летит быстро (больше 60 студов в сек)
        if projectile:IsA("BasePart") and projectile.Velocity.Magnitude > 60 then
            local vectorToMe = (HumanoidRoot.Position - projectile.Position)
            local distance = vectorToMe.Magnitude
            
            -- Если снаряд ближе 30 метров
            if distance < 30 then
                -- Главная математика: летит ли он ПРЯМО в нас?
                -- Используем Dot Product (скалярное произведение)
                local isComingAtUs = projectile.Velocity.Unit:Dot(vectorToMe.Unit)
                
                if isComingAtUs > 0.92 then -- 0.92 = очень высокая точность наведения на нас
                    -- МГНОВЕННЫЙ УКЛОН (Матрица)
                    -- Находим вектор "вбок" относительно летящей пули
                    local sideVector = projectile.Velocity.Unit:Cross(Vector3.new(0, 1, 0)).Unit
                    
                    -- Телепортируем CFrame в сторону на 12 единиц
                    HumanoidRoot.CFrame = HumanoidRoot.CFrame * CFrame.new(sideVector * 12)
                    
                    -- Визуальный эффект "тени" при уклонении
                    local ghost = Instance.new("Part")
                    ghost.Size, ghost.CFrame, ghost.Anchored = Character["Left Leg"].Size, HumanoidRoot.CFrame, true
                    ghost.CanCollide, ghost.Transparency, ghost.Color = false, 0.5, Color3.fromRGB(0, 255, 255)
                    ghost.Parent = workspace
                    Debris:AddItem(ghost, 0.3)
                    
                    table.insert(notificationQueue, "Smart Dodge: Уклонение!")
                end
            end
        end
    end
end)

Как проверить, что всё работает:
 * Интерфейс: В твоем меню должны появиться две кнопки с иконками 🎯 и 🎭.
 * Silent Aim: Возьми любое оружие в игре (автомат, пистолет) и стреляй примерно в сторону врага. Если он в радиусе и его видно — пули сами полетят в голову.
 * Smart Dodge: Попроси кого-нибудь выстрелить в тебя или встань перед турелью. Твой персонаж должен «отскакивать» в сторону за мгновение до попадания.
Почему это круче обычных читов?
 * Silent Aim не поворачивает твою камеру. Ты можешь смотреть в одну сторону, а стрелять в другую — это выглядит гораздо естественнее для наблюдателей.
 * Dodge использует Dot Product. Это значит, что если кто-то стреляет рядом с тобой, чит не будет дергаться зря. Он сработает только тогда, когда попадание неизбежно.
