Поскольку кнопки теперь создаются автоматически через таблицу, тебе осталось только вставить «мозги» этих функций в самый конец твоего скрипта lua.lua (перед disableAll).
Копируй этот блок целиком:
-- ==========================================
-- LOGIC: GOD-TIER COMBAT (Silent Aim & Smart Dodge)
-- ==========================================

-- Поиск цели для Сайлент Аима
local function getBestTarget()
    local target = nil
    local dist = 1000
    for _, p in pairs(Players:GetPlayers()) do
        if p ~= LocalPlayer and p.Character and p.Character:FindFirstChild("Humanoid") and p.Character.Humanoid.Health > 0 then
            if p.Team ~= LocalPlayer.Team then -- Проверка на команду
                local head = p.Character:FindFirstChild("Head")
                if head then
                    local mag = (head.Position - HumanoidRoot.Position).Magnitude
                    if mag < dist then
                        -- Проверка видимости (через стены не стреляем для беспалевности)
                        local ray = Ray.new(Camera.CFrame.Position, (head.Position - Camera.CFrame.Position).Unit * mag)
                        local part = workspace:FindPartOnRayWithIgnoreList(ray, {Character, workspace.CurrentCamera})
                        if part and part:IsDescendantOf(p.Character) then
                            target = head
                            dist = mag
                        end
                    end
                end
            end
        end
    end
    return target
end

-- SILENT AIM: Взлом системного выстрела (Metatable Hook)
local oldNamecall
oldNamecall = hookmetamethod(game, "__namecall", function(self, ...)
    local method = getnamecallmethod()
    if Settings.Features.SilentAim and (method == "Raycast" or method == "FindPartOnRayWithIgnoreList") then
        local t = getBestTarget()
        if t then return t, t.Position, t.CFrame.LookVector, t.Material end
    end
    return oldNamecall(self, ...)
end)

-- SMART DODGE: Автоматическое уклонение (Heartbeat)
RunService.Heartbeat:Connect(function()
    if not Settings.Features.SmartDodge or not HumanoidRoot then return end
    
    for _, obj in pairs(workspace:GetChildren()) do
        if obj:IsA("BasePart") and obj.Velocity.Magnitude > 60 then
            local toMe = (HumanoidRoot.Position - obj.Position)
            if toMe.Magnitude < 25 then
                -- Если пуля летит прямо в нас (математический расчет вектора)
                if obj.Velocity.Unit:Dot(toMe.Unit) > 0.9 then
                    -- Рывок в сторону
                    local side = obj.Velocity.Unit:Cross(Vector3.new(0,1,0)).Unit
                    HumanoidRoot.CFrame = HumanoidRoot.CFrame * CFrame.new(side * 10)
                    
                    -- Маленькое уведомление, что мы увернулись
                    table.insert(notificationQueue, "Dodge: Уклонение выполнено!")
                end
            end
        end
    end
end)
